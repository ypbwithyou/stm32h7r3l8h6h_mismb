
子板协议

校准{
    uint8 bitflag，    //0-2bit为为对应通道1-3，如果bitflag全0则认为是禁用dac，全1认为统一设一个值，后续float给一个给三个都行
    float voltage0，//各通道电压值
    float voltage1，
    float voltage2
}

桥路{
    uint8 exc_en    //激励，0失能，1使能
    uint8 bridge    //bit0-2，通道0-2的电压信号来源，0全桥，1半桥，
    uint8 bridgeShunt    //bit0-2，通道0-2接如分流电阻，0不接，1接入
}

增益{
    uint8 gain,    //bit0-2分别表示三通道，0代表1倍，1代表10倍
    uint16 pga,    //bit0-2，代表pga0_0-pga0_2，全0不放大，全1放大128倍，同样bit3-5为pga1，bit6-8为pga2
}
